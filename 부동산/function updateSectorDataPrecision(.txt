function updateSectorDataPrecision() {
  const SLIDE_ID = "1AvMFgY6uFNJ0pUn8EWaW-YQWyaByGVrgYDFpivesJUc"; 
  
  try {
    const presentation = SlidesApp.openById(SLIDE_ID);
    const slide = presentation.getSlides()[0];
    const allShapes = slide.getShapes();
    
    let sensors = {};
    let drawings = [];

    // 1. 분류 로직 개선: 이름에 'area'가 있으면 무조건 센서, 없으면 무조건 분석 도형
    allShapes.forEach(shape => {
      let title = shape.getTitle() || "";
      let lowerTitle = title.toLowerCase().replace(/\s/g, "");
      
      if (lowerTitle.includes('area')) {
        sensors[lowerTitle] = shape;
      } else {
        // 캡처 이미지(이미지 객체)는 스크립트가 getShapes()에서 자동으로 제외하기도 하지만,
        // 안전하게 사용자가 직접 그린 '도형'만 수집합니다.
        if (shape.getShapeType() != null) {
          drawings.push(shape);
        }
      }
    });

    let finalResults = [];

    // 2. 9개 섹터별 면적 계산
    for (let i = 1; i <= 9; i++) {
      let sensor = sensors['area' + i];
      if (!sensor) { 
        finalResults.push(["0.0%"]); 
        continue; 
      }

      let sL = sensor.getLeft(), sT = sensor.getTop();
      let sW = sensor.getWidth(), sH = sensor.getHeight();
      let sR = sL + sW, sB = sT + sH;
      
      let overlappingArea = 0;

      drawings.forEach(dwg => {
        let dL = dwg.getLeft(), dT = dwg.getTop();
        let dW = dwg.getWidth(), dH = dwg.getHeight();
        let dR = dL + dW, dB = dT + dH;

        // 교집합 면적 계산
        let xOverlap = Math.max(0, Math.min(sR, dR) - Math.max(sL, dL));
        let yOverlap = Math.max(0, Math.min(sB, dB) - Math.max(sT, dT));
        
        if (xOverlap > 0 && yOverlap > 0) {
          overlappingArea += (xOverlap * yOverlap);
        }
      });

      let ratio = (overlappingArea / (sW * sH)) * 100;
      finalResults.push([ Math.min(ratio, 100).toFixed(1) + "%" ]);
    }

    // 3. 시트에 결과 기록
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    let sheet = ss.getSheetByName("분석결과") || ss.getActiveSheet();
    sheet.getRange("B2:B10").setValues(finalResults);

    // 에러 방지를 위해 로그만 남기고 종료 (성공 시 시트 값이 바뀝니다)
    Logger.log("분석 완료");

  } catch (e) {
    Logger.log("오류 발생: " + e.message);
  }
}

function onOpen() {
  const ui = SpreadsheetApp.getUi();
  ui.createMenu('🚀 PPT 자동화')
      .addItem('3x3 섹터 데이터 업데이트', 'updateSectorDataPrecision')
      .addToUi();
}
